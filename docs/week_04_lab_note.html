    SELECT AVG(dep_delay) AS avg_dep_delay
    FROM flights
    WHERE origin = 'EWR'

SELECT DISTINCT a.name AS carrier_name
FROM flights f
JOIN avg_dep_delay_ewr avg ON f.dep_delay > avg.avg_dep_delay
